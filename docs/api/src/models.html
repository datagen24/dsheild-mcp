 59class SecurityEvent(BaseModel):
 60    """Model for security events from DShield SIEM."""
 61    
 62    id: str = Field(..., description="Unique event identifier")
 63    timestamp: datetime = Field(..., description="Event timestamp")
 64    source_ip: Optional[str] = Field(None, description="Source IP address")
 65    destination_ip: Optional[str] = Field(None, description="Destination IP address")
 66    source_port: Optional[int] = Field(None, description="Source port")
 67    destination_port: Optional[int] = Field(None, description="Destination port")
 68    protocol: Optional[str] = Field(None, description="Network protocol")
 69    event_type: str = Field(..., description="Type of security event")
 70    severity: EventSeverity = Field(EventSeverity.MEDIUM, description="Event severity")
 71    category: EventCategory = Field(EventCategory.OTHER, description="Event category")
 72    description: str = Field(..., description="Event description")
 73    
 74    # DShield-specific fields
 75    country: Optional[str] = Field(None, description="Country of origin")
 76    asn: Optional[str] = Field(None, description="Autonomous System Number")
 77    organization: Optional[str] = Field(None, description="Organization name")
 78    reputation_score: Optional[float] = Field(None, description="DShield reputation score")
 79    attack_count: Optional[int] = Field(None, description="Number of attacks from this source")
 80    first_seen: Optional[datetime] = Field(None, description="First seen timestamp")
 81    last_seen: Optional[datetime] = Field(None, description="Last seen timestamp")
 82    tags: List[str] = Field(default_factory=list, description="DShield tags")
 83    attack_types: List[str] = Field(default_factory=list, description="Types of attacks")
 84    
 85    # Metadata
 86    raw_data: Dict[str, Any] = Field(default_factory=dict, description="Raw event data")
 87    indices: List[str] = Field(default_factory=list, description="Source indices")
 88    
 89    @field_validator('source_ip', 'destination_ip')
 90    @classmethod
 91    def validate_ip_address(cls, v):
 92        """Validate IP address format."""
 93        if v is not None:
 94            try:
 95                ipaddress.ip_address(v)
 96            except ValueError:
 97                raise ValueError(f"Invalid IP address: {v}")
 98        return v
 99    
100    @field_validator('source_port', 'destination_port')
101    @classmethod
102    def validate_port(cls, v):
103        """Validate port number."""
104        if v is not None and (v < 1 or v > 65535):
105            raise ValueError(f"Invalid port number: {v}")
106        return v
107    
108    @field_validator('reputation_score')
109    @classmethod
110    def validate_reputation_score(cls, v):
111        """Validate reputation score range."""
112        if v is not None and (v < 0 or v > 100):
113            raise ValueError(f"Reputation score must be between 0 and 100: {v}")
114        return v

153    def _load_user_config_file(self) -> Dict[str, Any]:
154        """Load user configuration from file."""
155        user_config_paths = [
156            Path("user_config.yaml"),
157            Path("config/user_config.yaml"),
158            Path.home() / ".dshield-mcp" / "user_config.yaml"
159        ]
160        
161        for config_path in user_config_paths:
162            if config_path.exists():
163                try:
164                    with open(config_path, 'r') as f:
165                        config = yaml.safe_load(f)
166                    logger.info(f"Loaded user config from: {config_path}")
167                    return config or {}
168                except Exception as e:
169                    logger.warning(f"Failed to load user config from {config_path}: {e}")

319        """Validate all settings and log warnings for invalid values."""
320        errors = []
321        warnings = []
322        
323        # Query Settings Validation
324        if self.query_settings.default_page_size > self.query_settings.max_page_size:

326        if self.query_settings.default_timeout_seconds > self.query_settings.max_timeout_seconds:
327            errors.append("default_timeout_seconds cannot be greater than max_timeout_seconds")
328        if self.query_settings.fallback_strategy not in ["aggregate", "sample", "error"]:
329            errors.append("fallback_strategy must be one of: aggregate, sample, error")
330        
331        # Pagination Settings Validation
332        if self.pagination_settings.default_method not in ["page", "cursor"]:
333            errors.append("pagination_method must be one of: page, cursor")
334        if self.pagination_settings.max_pages_per_request <= 0:
335            errors.append("max_pages_per_request must be positive")
336        
337        # Streaming Settings Validation
338        if self.streaming_settings.default_chunk_size > self.streaming_settings.max_chunk_size:
339            errors.append("default_chunk_size cannot be greater than max_chunk_size")
340        if self.streaming_settings.session_timeout_minutes <= 0:
341            errors.append("session_timeout_minutes must be positive")
342        
343        # Performance Settings Validation
344        if self.performance_settings.cache_ttl_seconds <= 0:
345            errors.append("cache_ttl_seconds must be positive")
346        if self.performance_settings.max_cache_size <= 0:
347            errors.append("max_cache_size must be positive")
348        if self.performance_settings.connection_pool_size <= 0:
349            errors.append("connection_pool_size must be positive")
350        
351        # Security Settings Validation
352        if self.security_settings.rate_limit_requests_per_minute <= 0:
353            errors.append("rate_limit_requests_per_minute must be positive")
354        if self.security_settings.max_query_results <= 0:
355            errors.append("max_query_results must be positive")
356        
357        # Logging Settings Validation
358        valid_log_levels = ["DEBUG", "INFO", "WARNING", "ERROR", "CRITICAL"]
359        if self.logging_settings.log_level.upper() not in valid_log_levels:
360            errors.append(f"log_level must be one of: {', '.join(valid_log_levels)}")
361        if self.logging_settings.log_format not in ["json", "text"]:
362            errors.append("log_format must be one of: json, text")
363        if self.logging_settings.max_log_size_mb <= 0:
364            errors.append("max_log_size_mb must be positive")
365        
366        # Campaign Settings Validation
367        if self.campaign_settings.correlation_window_minutes <= 0:
368            errors.append("correlation_window_minutes must be positive")
369        if not 0.0 <= self.campaign_settings.min_confidence_threshold <= 1.0:
370            errors.append("min_confidence_threshold must be between 0.0 and 1.0")
371        if self.campaign_settings.max_campaign_events <= 0:
372            errors.append("max_campaign_events must be positive")
373        if self.campaign_settings.max_expansion_depth <= 0:
374            errors.append("max_expansion_depth must be positive")
375        if self.campaign_settings.expansion_timeout_seconds <= 0:
376            errors.append("expansion_timeout_seconds must be positive")
377        
378        # Log errors and warnings
379        if errors:
380            raise ValueError(f"Configuration validation errors: {'; '.join(errors)}")
381        
382        if warnings:
383            for warning in warnings:
384                logger.warning(warning)

320        """Validate all settings and log warnings for invalid values."""
321        errors = []
322        warnings = []
323        
324        # Query Settings Validation
325        if self.query_settings.default_page_size > self.query_settings.max_page_size:

330            errors.append("fallback_strategy must be one of: aggregate, sample, error")
331        
332        # Pagination Settings Validation
333        if self.pagination_settings.default_method not in ["page", "cursor"]:
334            errors.append("pagination_method must be one of: page, cursor")
335        if self.pagination_settings.max_pages_per_request <= 0:
336            errors.append("max_pages_per_request must be positive")
337        
338        # Streaming Settings Validation
339        if self.streaming_settings.default_chunk_size > self.streaming_settings.max_chunk_size:

342            errors.append("session_timeout_minutes must be positive")
343        
344        # Performance Settings Validation

346            errors.append("cache_ttl_seconds must be positive")
347        if self.performance_settings.max_cache_size <= 0:
348            errors.append("max_cache_size must be positive")
349        if self.performance_settings.connection_pool_size <= 0:
350            errors.append("connection_pool_size must be positive")
351        
352        # Security Settings Validation

354            errors.append("rate_limit_requests_per_minute must be positive")
355        if self.security_settings.max_query_results <= 0:
356            errors.append("max_query_results must be positive")
357

361            errors.append(f"log_level must be one of: {', '.join(valid_log_levels)}")
362        if self.logging_settings.log_format not in ["json", "text"]:
363            errors.append("log_format must be one of: json, text")
364        if self.logging_settings.max_log_size_mb <= 0:
365            errors.append("max_log_size_mb must be positive")
366        
367        # Campaign Settings Validation
368        if self.campaign_settings.correlation_window_minutes <= 0:
369            errors.append("correlation_window_minutes must be positive")
370        if not 0.0 <= self.campaign_settings.min_confidence_threshold <= 1.0:
371            errors.append("min_confidence_threshold must be between 0.0 and 1.0")
372        if self.campaign_settings.max_campaign_events <= 0:
373            errors.append("max_campaign_events must be positive")
374        if self.campaign_settings.max_expansion_depth <= 0:

376        if self.campaign_settings.expansion_timeout_seconds <= 0:
377            errors.append("expansion_timeout_seconds must be positive")
378        
379        # Log errors and warnings
380        if errors:
381            raise ValueError(f"Configuration validation errors: {'; '.join(errors)}")
382        
383        if warnings:
384            for warning in warnings:
385                logger.warning(warning)

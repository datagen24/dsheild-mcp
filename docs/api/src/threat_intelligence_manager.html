Threat Intelligence Manager for DShield MCP.

This module provides a centralized manager for coordinating multiple threat intelligence sources including DShield, VirusTotal, Shodan, and other security APIs. It handles source coordination, rate limiting, caching, and result correlation.

Features:

  • Multi-source threat intelligence aggregation
  • Intelligent correlation and scoring
  • Advanced caching with configurable TTL
  • Rate limiting and error handling
  • Extensible architecture for new sources

Example:

from src.threat_intelligence_manager import ThreatIntelligenceManager async with ThreatIntelligenceManager() as manager: ... result = await manager.enrich_ip_comprehensive("8.8.8.8") ... print(result.overall_threat_score)

210    async def enrich_ip_comprehensive(self, ip_address: str) -> ThreatIntelligenceResult:
211        """Comprehensive IP enrichment from multiple sources.
212        
213        Queries all enabled threat intelligence sources for information about
214        the specified IP address, aggregates results, and provides a comprehensive
215        threat assessment.
216        
217        Args:
218            ip_address: The IP address to enrich
219            
220        Returns:
221            ThreatIntelligenceResult: Comprehensive threat intelligence data
222            
223        Raises:
224            ValueError: If IP address is invalid
225            RuntimeError: If no sources are available
226        """
227        # Validate IP address
228        try:
229            import ipaddress
230            ipaddress.ip_address(ip_address)
231        except ValueError:
232            raise ValueError(f"Invalid IP address: {ip_address}")
233        
234        # Check cache first
235        cache_key = f"comprehensive_ip_{ip_address}"
236        
237        # Check SQLite cache first (if enabled)
238        if self.sqlite_cache_enabled:
239            cached_result = await self._get_sqlite_cached_result(cache_key)
240            if cached_result:
241                logger.debug("Returning SQLite cached IP enrichment result", ip_address=ip_address)
242                cached_result.cache_hit = True
243                return cached_result
244        
245        # Check memory cache as fallback
246        cached_result = self._get_cached_result(cache_key)
247        if cached_result:
248            logger.debug("Returning memory cached IP enrichment result", ip_address=ip_address)
249            cached_result.cache_hit = True
250            return cached_result
251        
252        if not self.clients:
253            raise RuntimeError("No threat intelligence sources available")
254        
255        logger.info("Starting comprehensive IP enrichment", 
256                   ip_address=ip_address,
257                   available_sources=list(self.clients.keys()))
258        
259        # Query all enabled sources concurrently
260        tasks = []
261        for source, client in self.clients.items():
262            if hasattr(client, 'get_ip_reputation'):
263                tasks.append(self._query_source_async(source, client, ip_address))
264        
265        # Wait for all queries to complete
266        source_results = await asyncio.gather(*tasks, return_exceptions=True)
267        
268        # Process results
269        result = ThreatIntelligenceResult(ip_address=ip_address)
270        successful_sources = []
271        
272        for source, source_result in zip(self.clients.keys(), source_results):
273            if isinstance(source_result, Exception):
274                logger.warning("Source query failed", 
275                              source=source, 
276                              ip_address=ip_address, 
277                              error=str(source_result))
278                continue
279            
280            result.source_results[source] = source_result
281            successful_sources.append(source)
282        
283        result.sources_queried = successful_sources
284        
285        # Correlate and score results
286        await self._correlate_results(result)
287        
288        # Cache the result in both SQLite and memory
289        if self.sqlite_cache_enabled:
290            self._cache_result_to_sqlite(cache_key, result)
291        self._cache_result(cache_key, result)
292        
293        logger.info("IP enrichment completed", 
294                   ip_address=ip_address,
295                   sources_queried=len(successful_sources),
296                   overall_threat_score=result.overall_threat_score,
297                   confidence_score=result.confidence_score)
298        
299        return result

427    async def _correlate_results(self, result: ThreatIntelligenceResult) -> None:
428        """Correlate results from multiple sources.
429        
430        Analyzes results from different sources and calculates aggregated
431        threat scores and confidence levels.
432        
433        Args:
434            result: The threat intelligence result to correlate
435        """
436        if not result.source_results:
437            logger.warning("No source results to correlate")
438            return
439        
440        # Calculate overall threat score
441        threat_scores = []
442        confidence_scores = []
443        
444        for source, source_data in result.source_results.items():
445            if isinstance(source_data, dict):
446                # Extract threat score
447                if 'threat_score' in source_data:
448                    threat_scores.append(float(source_data['threat_score']))
449                elif 'reputation_score' in source_data and source_data['reputation_score'] is not None:
450                    # Convert reputation score to threat score (inverse relationship)
451                    rep_score = float(source_data['reputation_score'])
452                    threat_score = 100 - rep_score  # Higher reputation = lower threat
453                    threat_scores.append(threat_score)
454                
455                # Extract confidence score
456                if 'confidence' in source_data:
457                    confidence_scores.append(float(source_data['confidence']))
458                else:
459                    # Default confidence based on source
460                    default_confidence = {
461                        ThreatIntelligenceSource.DSHIELD: 0.8,
462                        ThreatIntelligenceSource.VIRUSTOTAL: 0.9,
463                        ThreatIntelligenceSource.SHODAN: 0.7
464                    }.get(source, 0.5)
465                    confidence_scores.append(default_confidence)
466        
467        # Calculate weighted averages
468        if threat_scores:
469            result.overall_threat_score = sum(threat_scores) / len(threat_scores)
470        
471        if confidence_scores:
472            result.confidence_score = sum(confidence_scores) / len(confidence_scores)
473        
474        # Correlate threat indicators
475        all_indicators = []
476        for source_data in result.source_results.values():
477            if isinstance(source_data, dict):
478                if 'indicators' in source_data:
479                    all_indicators.extend(source_data['indicators'])
480                elif 'attack_types' in source_data:
481                    all_indicators.extend(source_data['attack_types'])
482                elif 'tags' in source_data:
483                    all_indicators.extend(source_data['tags'])
484        
485        # Remove duplicates and correlate
486        result.threat_indicators = self._deduplicate_indicators(all_indicators)
487        
488        # Aggregate geographic and network data
489        self._aggregate_geographic_data(result)
490        self._aggregate_network_data(result)
491        
492        # Determine timestamps
493        self._determine_timestamps(result)

495    def _deduplicate_indicators(self, indicators: List[Any]) -> List[Dict[str, Any]]:
496        """Remove duplicate indicators and merge metadata.
497

503        """

507        # Convert to strings for deduplication

509        unique_indicators = list(set(indicator_strings))
510        
511        # Create structured indicator objects
512        structured_indicators = []
513        for indicator in unique_indicators:
514            structured_indicators.append({

516                "type": self._classify_indicator(indicator),
517                "count": indicator_strings.count(indicator),
518                "sources": len(self.clients)  # All sources contributed
519            })
520        
521        return structured_indicators

523    def _classify_indicator(self, indicator: str) -> str:
524        """Classify the type of indicator.
525        
526        Args:
527            indicator: The indicator string
528            
529        Returns:
530            The indicator type classification
531        """
532        indicator_lower = indicator.lower()
533        
534        # IP address patterns
535        if any(char.isdigit() for char in indicator_lower) and '.' in indicator_lower:
536            return "ip_address"
537        
538        # Domain patterns
539        if '.' in indicator_lower and not any(char.isdigit() for char in indicator_lower.split('.')[0]):
540            return "domain"
541        
542        # Hash patterns
543        if len(indicator_lower) in [32, 40, 64] and all(c in '0123456789abcdef' for c in indicator_lower):
544            return "hash"
545        
546        # CVE patterns
547        if indicator_lower.startswith('cve-'):
548            return "cve"
549        
550        # Default to generic
551        return "generic"

622

628        
629        # Get rate limit configuration

633        
634        if len(tracker) >= rate_limit:
635            raise RuntimeError(f"Rate limit exceeded for {source.value}")
636        
637        tracker.append(current_time)
